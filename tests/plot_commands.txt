# Climate Change Subplots Commands

# Plot 1: CO2 concentrations (Mauna Loa & South Pole) in separate subplots
mfp_multi_plot(
    commands="climet_change_subplots.csv using 1:0 with lines lc blue title "Mauna Loa CO2" xlabel "Date" ylabel "CO2 (ppm)", climet_change_subplots.csv using 1:2 with lines lc red title "South Pole CO2" xlabel "Date" ylabel "CO2 (ppm)"",
    save="climet_co2_subplots.png",
    subplot_layout="AB",
    title="CO2 Concentration Over Time"
)

# Plot 2: Heat content with ±2σ error bands
mfp_multi_plot(
    commands="climet_change_subplots.csv using 1:3 with errorshade yerr_col 5:6 lc steelblue title "Heat Content", climet_change_subplots.csv using 1:3 with lines lc steelblue title "Heat Content Mean"",
    save="climet_heat_content.png",
    subplot_layout="AB",
    title="Heat Content with ±2σ",
    xlabel="Date",
    ylabel="Heat Content (W/m²)"
)

# Plot 3: Temperature anomaly with ±2σ error bands
mfp_multi_plot(
    commands="climet_change_subplots.csv using 2:9 with errorshade yerr_col 8:10 lc tomato title "Temp Anomaly ±2σ", climet_change_subplots.csv using 2:9 with lines lc tomato title "Temp Anomaly Mean"",
    save="climet_temp_anomaly.png",
    subplot_layout="AB",
    title="Temperature Anomaly with ±2σ",
    xlabel="Date",
    ylabel="Temperature Anomaly (°C)"
)

# Plot 4: All 4 metrics in 2x2 grid
mfp_multi_plot(
    commands="climet_change_subplots.csv using 1:0 with lines lc blue title "Mauna Loa CO2" xlabel "Date" ylabel "CO2 (ppm)", climet_change_subplots.csv using 1:2 with lines lc red title "South Pole CO2" xlabel "Date" ylabel "CO2 (ppm)", climet_change_subplots.csv using 1:3 with errorshade yerr_col 5:6 lc steelblue title "Heat Content ±2σ" xlabel "Date" ylabel "Heat Content", climet_change_subplots.csv using 2:9 with errorshade yerr_col 8:10 lc tomato title "Temp Anomaly ±2σ" xlabel "Date" ylabel "Temp Anomaly"",
    save="climate_change_all_subplots.png",
    subplot_layout="AB-CD",
    title="Climate Change Indicators"
)
